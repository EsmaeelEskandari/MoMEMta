34 MoMEMta::MoMEMta(const Configuration& configuration) {
35 
36  // Initialize shared memory pool for modules
37  m_pool.reset(new Pool());
38 
39  // Create phase-space points vector, input for many modules
40  m_pool->current_module("cuba");
41  m_ps_points = m_pool->put<std::vector<double>>({"cuba", "ps_points"});
42  m_ps_weight = m_pool->put<double>({"cuba", "ps_weight"});
43 
44  // Create vector for input particles
45  m_pool->current_module("input");
46  m_particles = m_pool->put<std::vector<LorentzVector>>({"input", "particles"});
47  // Create input for met
48  m_met = m_pool->put<LorentzVector>({"input", "met"});
49 
50  // Construct modules from configuration
51  std::vector<Configuration::Module> light_modules = configuration.getModules();
52  for (const auto& module: light_modules) {
53  m_pool->current_module(module.name);
54  m_modules.push_back(ModuleFactory::get().create(module.type, m_pool, module.parameters));
55  m_modules.back()->configure();
56  }
57 
58  m_n_dimensions = 0;
59  for (const auto& module: m_modules) {
60  m_n_dimensions += module->dimensions();
61  }
62 
63  LOG(info) << "Number of dimensions for integration: " << m_n_dimensions;
64 
65  // Resize pool ps-points vector
66  m_ps_points->resize(m_n_dimensions);
67 
68  // Find which modules produces the 'integrands' output, which defines the integrand.
69  // **There can be only one!**
70  m_pool->current_module("momemta");
71  bool foundIntegrands = false;
72  for (const auto& module: m_modules) {
73  if(m_pool->exists({module->name(), "integrands"})) {
74  if(!foundIntegrands){
75  m_integrands = m_pool->get<std::vector<double>>({module->name(), "integrands"});
76  foundIntegrands = true;
77  LOG(debug) << "Module " << module->name() << " produces the integrand.";
78  } else {
79  throw integrands_output_error("Only one module can produce the `integrands` output.");
80  }
81  }
82  }
83  if(!foundIntegrands)
84  throw integrands_output_error("No module found which produces the mandatory `integrands` output.");
85 
86  m_cuba_configuration = configuration.getCubaConfiguration();
87 
88  const Pool::DescriptionMap& description = m_pool->description();
89  graph::build(description, m_modules, [&description, this](const std::string& module) {
90  // Clean the pool for each removed module
91  const Description& d = description.at(module);
92  for (const auto& input: d.inputs)
93  this->m_pool->remove_if_invalid(input);
94  for (const auto& output: d.outputs)
95  this->m_pool->remove({module, output});
96  });
97 
98  // Freeze the pool after removing unneeded modules
99  m_pool->freeze();
100 
101  cubacores(0, 0);
102 }

106  module->finish();
107  }

133  int64_t seed = m_cuba_configuration.get<int64_t>("seed", 0);

143  int64_t grid_number = m_cuba_configuration.get<int64_t>("grid_number", 0);

150  llVegas(
151  m_n_dimensions, // (int) dimensions of the integrated volume

156  relative_accuracy, // (double) requested relative accuracy /

159  seed, // (int) seed (seed==0 => SOBOL; seed!=0 && control flag "level"==0 => Mersenne Twister)
160  min_eval, // (int) minimum number of integrand evaluations
161  max_eval, // (int) maximum number of integrand evaluations (approx.!)
162  n_start, // (int) number of integrand evaluations per interations (to start)
163  n_increase, // (int) increase in number of integrand evaluations per interations
164  batch_size, // (int) batch size for sampling
165  grid_number, // (int) grid number, 1-10 => up to 10 grids can be stored, and re-used for other integrands (provided they are not too different)
166  grid_file.c_str(), // (char*) name of state file => state can be stored and retrieved for further refinement
167  NULL, // (int*) "spinning cores": -1 || NULL <=> integrator takes care of starting & stopping child processes (other value => keep or retrieve child processes, probably not useful here)
168  &neval, // (int*) actual number of evaluations done

171  &error, // (double*) integration error ([ncomp])
172  &prob // (double*) Chi-square p-value that error is not reliable (ie should be <0.95) ([ncomp])
173  );
174 
175  for (const auto& module: m_modules) {

179  return std::vector<std::pair<double, double>>({{mcResult, error}});

182 double MoMEMta::integrand(const double* psPoints, const double* weights) {
183 
184  // Store phase-space points into the pool
185  std::memcpy(m_ps_points->data(), psPoints, sizeof(double) * m_n_dimensions);
186 
187  // Store phase-space weight into the pool
188  *m_ps_weight = *weights;
189 
190  for (auto& module: m_modules) {
191  module->work();
192  }
193 
194  double sum = 0;
195  for (const auto& p: *m_integrands) {
196  sum += p;
197  }
198 
199  return sum;
200 }
