113  virtual void work() override {
114 
115  const double& ps_point = m_ps_point.get<double>();
116  const LorentzVector& reco_particle = m_input.get<LorentzVector>();
117 
118  const double rec_E = reco_particle.E();
119  const double range = GetDeltaRange(rec_E);
120  const double gen_E = rec_E - GetDeltaMax(rec_E) + range*ps_point;
121  const double delta = rec_E - gen_E;
122 
123  // To change the particle's energy without changing its direction and mass:
124  double gen_pt = std::sqrt(SQ(gen_E) - SQ(reco_particle.M())) / std::cosh(reco_particle.Eta());
125  output->SetCoordinates(
126  gen_pt * std::cos(reco_particle.Phi()),
127  gen_pt * std::sin(reco_particle.Phi()),
128  gen_pt * std::sinh(reco_particle.Eta()),
129  gen_E);
130 
131  // The bin number is a ROOT "global bin number" using a 1D representation of the TH2
132  const int bin = m_th2->FindFixBin(std::min(gen_E, m_fallBackEgenMax), delta);
133  // Compute TF*jacobian, where the jacobian includes the transformation of [0,1]->[range_min,range_max] and d|P|/dE
134  *TF_times_jacobian = m_th2->GetBinContent(bin) * range * dP_over_dE(*output);
135  }
